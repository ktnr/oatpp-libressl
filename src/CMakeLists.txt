
add_library(${OATPP_THIS_MODULE_NAME}

        oatpp-libressl/Config.cpp
        oatpp-libressl/Config.hpp
        oatpp-libressl/Connection.cpp
        oatpp-libressl/Connection.hpp

)

        CXX_STANDARD_REQUIRED ON
)
